268def tune_hyperparams(estimator_name, estimator, estimator_params, train_labels, train_images): 

269 """ 

270 Perform hyperparameter tuning. 

271 """ 

272 

273 # print(train_images) 

274 # print(train_labels) 

275 k_fold = KFold(n_splits=3) 

276 

277 best_model_estimator = Pipeline([("pca", PCA()), (estimator_name, estimator)]) 

278 sum_scores = 0 

279 best_model_score = 0.0 

280 # print(len(train_labels)) 

281 # print(len(train_images)) 

282 vector = np.vectorize(np.int16) 

283 

284 for train_index, test_index in k_fold.split(train_images): 

285 # print(train_index) 

286 xtrain, xtest = train_images[vector(train_index)], train_images[vector(test_index)] 

287 ytrain, ytest = train_labels[vector(train_index)], train_labels[vector(test_index)] 

288 

289 pipe = Pipeline([("pca", PCA()), (estimator_name, estimator)]) 

290 search = GridSearchCV(pipe, estimator_params, cv=5, return_train_score=True, n_jobs=-1, 

291 verbose=2, scoring="f1_macro") 

292 

293 search.fit(xtrain, ytrain) 

294 

295 est = search.best_estimator_ 

296 # print("ESST", est) 

297 # print("BESST", search.best_score_) 

298 

299 est_pipe = make_pipeline(est) 

300 est_pipe.fit(xtrain, ytrain) 

301 prediction = est_pipe.predict(xtest) 

302 

303 f1_score_est = f1_score(ytest, prediction, average="macro") 

304 # print("f1_score_: ", f1_score_est) 

305 

306 if f1_score_est > best_model_score: 

307 best_model_score = f1_score_est 

308 best_model_estimator = est 

309 

310 sum_scores = sum_scores + f1_score_est 

311 

312 mean_score = sum_scores / k_fold.get_n_splits() 

313 

314 print("Mean score:", mean_score, "\n") 

315 print("Best score:", best_model_score, "\n") 

316 print("Best estimator:", best_model_estimator) 

317 return mean_score, best_model_score, best_model_estimator
